DICE


This is a program based on the dice which we used to play board games
it basically gives output like dice by giving random numbers varying
from (1-6).
This program also has a short gui-like output which can let draw out
dices.
				

import random
key=input("ENTER ANY KEY TO ROLL A DIE OR PRESS 'ENTER' TO QUIT")
def first():
         space=' '
         column='|'
         number='0'
         top=space.ljust(21,'-')
         print(top)
         for i in range(3):
               print(column,space.rjust(18,' '),column)
         print(column,number.rjust(9,' '),column.rjust(10,' '))
         for i in range(3):
               print(column,space.rjust(18,' '),column)
         print(top)
def secondth():
         space=' '
         column='|'
         number='0'
         top=space.ljust(21,'-')
         print(top)
         print(column,number.rjust(1,' '),column.rjust(18,' '))
         for i in range(5):
            print(column,space.rjust(18,' '),column)
         print(column,number.rjust(18,' '),column.rjust(1,' '))
         print(top)
def thirdth():
         space=' '
         column='|'
         number='0'
         top=space.ljust(21,'-')
         print(top)
         print(column,number.rjust(1,' '),column.rjust(18,' '))
         for i in range(2):
            print(column,space.rjust(18,' '),column)
         print(column,number.rjust(9,' '),column.rjust(10,' '))
         for i in range(2):
            print(column,space.rjust(18,' '),column)
         print(column,number.rjust(18,' '),column.rjust(1,' '))
         print(top)
def four():
         space=' '
         column='|'
         number='0'
         top=space.ljust(21,'-')
         print(top)
         print(column,number.rjust(1,' '),number.rjust(16,' '),column.rjust(1,' '))
         for i in range(5):
            print(column,space.rjust(18,' '),column)
         print(column,number.rjust(1,' '),number.rjust(16,' '),column.rjust(1,' '))
         print(top)
def five():
         space=' '
         column='|'
         number='0'
         top=space.ljust(21,'-')
         print(top)
         print(column,number.rjust(1,' '),number.rjust(16,' '),column.rjust(1,' '))
         for i in range(2):
            print(column,space.rjust(18,' '),column)
         print(column,number.rjust(9,' '),column.rjust(10,' '))
         for i in range(2):
            print(column,space.rjust(18,' '),column)
         print(column,number.rjust(1,' '),number.rjust(16,' '),column.rjust(1,' '))
         print(top)
def sixth():
         space=' '
         column='|'
         number='0'
         top=space.ljust(21,'-')
         print(top)
         for i in range(2):
            print(column,number.rjust(1,' '),number.rjust(16,' '),column.rjust(1,' '))
            print(column,space.rjust(18,' '),column)
            print(column,space.rjust(18,' '),column)
         print(column,number.rjust(1,' '),number.rjust(16,' '),column.rjust(1,' '))
         print(top)
while key!='':
    s=random.randint(1,6)
    if s==1:
         first()
    if s==2:
         secondth()
    if s==3:
         thirdth()
    if s==4:
         four()
    if s==5:
         five()
    if s==6:
         sixth()
    key=input("IF YOU WANT TO PLAY AGAIN PRESS ANY KEY OR PRESS ENTER TO QUIT")

TIMER-STOPWATCH

This program is based on the stopwatch which we use in our everyday life
What this basically does is the stopwatch starts counting the seconds
When you press the start button.
Where as in Timer it makes it like a countdown
Like going from reverse like (10,9....,3,2,1)


import os
import time
space=' '
column='|'
def cls():
        os.system('cls')
key1=input("ENTER 'S' FOR STOPWATCH OR 'T' FOR TIMER OR PRESS 'ENTER' TO QUIT: ")
cls()
while key1 not in ['S','T','t','s','']:
    key1=input("ENTER A VALID INPUT,S FOR STOPWATCH OR T FOR TIMER:")
    cls()
if key1 in ['s','S']:
    def clock():
        hour =0
        minute=0
        second=0
        day=0
        while True:
  # FOR SPACING OF TOP PORTION
            for i in range(24):
                print(space.rjust(167,' '))

  # FOR PRINTING OF MATTER(CLOCK),PRINTING IT IN CENTER OF THE SCREEN
            print(space.rjust(80,' '),"THE TIME IS:", day ,':',
			hour , ":",  minute ,":", second)
            print(space.rjust(80,' '),' '.rjust(10,' '),"D   H   M   S")
            print(space.rjust(80,' '),"D=Day,H =Hour,M=Minute,S=Second")

  # FOR SPACING OF BOTTOM PORTION
            for i in range(24):
                print(space.rjust(167,' '))

            time.sleep(0.93)            #TO MAKE PRINTING DELAY BY 1 SEC
			#,WHICH WOULD MAKE IT MORE OF A PRACTICAL TIMER
            cls()                       #CALLING OF FUNCTION WHICH CAUSES
			#TO REPLACE IT IN ITSELF
            second +=1                  #FOR REDUCING SECOND BY 1

  #CONVERSION FACTORS OF SECOND,MINUTE,HOUR AND DAY
            if second == 60:
                  minute += 1
                  second = 0
            if minute == 60:
                  hour += 1
                  minute = 0
            if hour == 24:
                  day += 1
                  hour =0

    clock()

if key1 in ['t','T']:
    def clock():
        space = ' '

  # INITIALISING THE PARAMETERS OR INPUT
